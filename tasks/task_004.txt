# Details:
Create a styled Microsoft login button component:

```typescript
// src/components/auth/MicrosoftLoginButton.tsx
import React from 'react';
import { useMSALAuth } from '../../auth/MSALAuthContext';

interface MicrosoftLoginButtonProps {
  className?: string;
  onLoginSuccess?: () => void;
  onLoginError?: (error: string) => void;
}

const MicrosoftLoginButton: React.FC<MicrosoftLoginButtonProps> = ({
  className = '',
  onLoginSuccess,
  onLoginError
}) => {
  const { login, isLoading } = useMSALAuth();

  const handleLogin = async () => {
    try {
      await login();
      if (onLoginSuccess) onLoginSuccess();
    } catch (error: any) {
      if (onLoginError) onLoginError(error.message || 'Login failed');
    }
  };

  return (
    <button
      type="button"
      className={`microsoft-login-button ${className}`}
      onClick={handleLogin}
      disabled={isLoading}
      aria-label="Sign in with Microsoft"
    >
      <span className="microsoft-icon">
        {/* Microsoft logo SVG */}
        <svg xmlns="http://www.w3.org/2000/svg" width="21" height="21" viewBox="0 0 21 21">
          <rect x="1" y="1" width="9" height="9" fill="#f25022" />
          <rect x="1" y="11" width="9" height="9" fill="#00a4ef" />
          <rect x="11" y="1" width="9" height="9" fill="#7fba00" />
          <rect x="11" y="11" width="9" height="9" fill="#ffb900" />
        </svg>
      </span>
      <span className="button-text">
        {isLoading ? 'Signing in...' : 'Sign in with Microsoft'}
      </span>
    </button>
  );
};

export default MicrosoftLoginButton;
```
